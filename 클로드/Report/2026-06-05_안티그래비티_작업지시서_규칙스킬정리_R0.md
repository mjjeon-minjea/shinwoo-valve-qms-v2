**Step 2. 17종 이동** (메타13 + 스택무관2 + 리버스API2)
```powershell
$archive = @(
  'advanced-evaluation','evaluation','context-compression','context-optimization',
  'filesystem-context','harness-engineering','multi-agent-patterns','project-development',
  'tool-design','memory-systems','composition-patterns','mcp-builder','claude-api',
  'react-native-skills','baoyu-electron-extract',
  'baoyu-danger-gemini-web','notebooklm'
)

  $src = ".agent\skills\$s"
  if (Test-Path $src) { Move-Item -Path $src -Destination ".agent\_archive\skills\" -Force; Write-Host "moved: $s" }
  else { Write-Host "skip(없음): $s" }
}
```
