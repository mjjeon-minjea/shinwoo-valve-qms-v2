# QMS v2 배포 인프라 정비 Implementation Plan

**Goal:** 로컬-테스트웹-메인웹 3계층 배포 인프라를 정비하고, 원격 브랜치 분할 추적 설정 및 단방향 데이터 복제 자동화 스크립트를 구현하여 안전한 릴리즈 체계를 수립한다.

  - **[물리 차단 방어 코드 구현]**: 스크립트 작성 시 메인 DB를 초기화하는 클라이언트(`prodDb`)에는 쓰기 관련 호출 권한이 배제되도록 조회(`select`) 구문만 호출되게 제약하고, 데이터 삽입/갱신(CUD) 요청은 오직 테스트 DB 클라이언트(`stagingDb`)로만 하도록 강제하여 메인 DB 덮어쓰기 오작동을 원천 봉쇄합니다.
  - (완제 코드):
    ```javascript
    import { createClient } from '@supabase/supabase-js';
    import dotenv from 'dotenv';
    import path from 'path';
    import { fileURLToPath } from 'url';

    const __filename = fileURLToPath(import.meta.url);
    const __dirname = path.dirname(__filename);
    const rootDir = path.resolve(__dirname, '..');

    dotenv.config({ path: path.resolve(rootDir, '.env.local') });

    const prodUrl = "https://zuahpjdsypovxdplxryw.supabase.co"; 
    const prodKey = process.env.SUPABASE_SERVICE_ROLE_KEY; // 메인 DB 조회 권한
    const stagingUrl = "https://srzaanvojyhwzugoaimk.supabase.co";
    const stagingKey = process.env.SUPABASE_SERVICE_ROLE_KEY; // 테스트 DB 쓰기 권한

    if (!prodKey || !stagingKey) {
      console.error("❌ .env.local 환경변수 누락 (Service Role Key 필요)");
      process.exit(1);
    }

    // 🔒 메인 DB 읽기 전용으로만 사용 (CUD 메서드 호출을 일절 배제하여 보호)
    const prodDb = createClient(prodUrl, prodKey, {
      auth: { persistSession: false }
    });

    // 🔄 테스트 DB 쓰기 클라이언트
    const stagingDb = createClient(stagingUrl, stagingKey, {
      auth: { persistSession: false }
    });

    const TARGET_TABLES = [
      'users',
      'dev_notes',
      'notices',
      'weekly_reports',
      'process_inspections',
      'receiving_inspections',
      'system_settings'
    ];

    async function runClone() {
      console.log("🔄 [DB Clone] 메인 DB ➔ 테스트 DB 단방향 복제를 시작합니다.");
      
      for (const table of TARGET_TABLES) {
        console.log(`- Fetching ${table} from PROD...`);
        // 읽기(SELECT)만 실행
        const { data, error: fetchErr } = await prodDb.from(table).select('*');
        
        if (fetchErr) {
          console.error(`❌ [${table}] 메인 DB 조회 실패:`, fetchErr.message);
          continue;
        }

        if (!data || data.length === 0) {
          console.log(`  ℹ️ [${table}] 복사할 데이터 없음`);
          continue;
        }

        console.log(`- Upserting ${data.length} records into STAGING [${table}]...`);
        
        // 쓰기는 오직 stagingDb 객체만을 사용하여 반영
        const { error: upsertErr } = await stagingDb.from(table).upsert(data);
        if (upsertErr) {
          console.error(`❌ [${table}] 테스트 DB 반영 실패:`, upsertErr.message);
        } else {
          console.log(`  ✅ [${table}] 복제 및 동기화 완료 (${data.length}건)`);
        }
      }
      console.log("🎉 [DB Clone 완료] 메인 ➔ 테스트 복제 작업이 무결하게 종료되었습니다.");
    }

    runClone();
    ```
