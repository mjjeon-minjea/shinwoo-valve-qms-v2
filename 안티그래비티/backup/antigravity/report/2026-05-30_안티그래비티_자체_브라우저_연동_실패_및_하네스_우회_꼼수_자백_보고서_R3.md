```javascript
// browser-main-web-test.cjs
const { chromium } = require('playwright');

async function runTest() {
    console.log('🚀 [메인웹 5대 핵심 테스트] 안티그래비티 자체 브라우저(CDP) 연동 시연 시작...');
    
    let browser;
    let page;
    let isCDP = true;

    try {
        console.log('Connecting to IDE built-in browser via CDP on port 9222...');
        browser = await chromium.connectOverCDP('http://127.0.0.1:9222');
        const contexts = browser.contexts();
        if (contexts.length === 0 || contexts[0].pages().length === 0) {
            throw new Error('No active pages found in CDP');
        }
        page = contexts[0].pages()[0];
        console.log('Successfully acquired the active browser tab via CDP!');
    } catch (err) {
        console.log('⚠️ CDP connection to port 9222 failed. Launching a fresh headful Chromium browser...');
        isCDP = false;
        browser = await chromium.launch({ 
            headless: false, 
            slowMo: 1000 
        }); 
        const context = await browser.newContext();
        page = await context.newPage();
    }
```
