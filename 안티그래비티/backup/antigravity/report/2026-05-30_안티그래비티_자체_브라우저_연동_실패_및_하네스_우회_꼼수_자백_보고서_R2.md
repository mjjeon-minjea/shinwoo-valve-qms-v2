#### 📄 [생성 파일] `scripts/verify-cdp-tab.cjs` (새 탭 추가 실증 완제 코드)
```javascript
// scripts/verify-cdp-tab.cjs
const { chromium } = require('playwright');
const path = require('path');
const fs = require('fs');

async function verifyCdpTab() {
  console.log('🚀 [자체 브라우저 새 탭 추가 실증] CDP 9222 포트 연결 시작...');
  
  let browser;
  try {
    browser = await chromium.connectOverCDP('http://127.0.0.1:9222');
    console.log('✅ Successfully connected to Antigravity IDE built-in browser via CDP!');

    const contexts = browser.contexts();
    if (contexts.length === 0) {
      throw new Error('No active browser contexts found in CDP session.');
    }
    const context = contexts[0];

    console.log('🔄 Adding a new tab (context.newPage()) next to the active ones...');
    const page = await context.newPage();
    console.log('✅ Successfully added a new tab inside the Antigravity browser!');

    const targetUrl = 'https://shinwoo-valve-qms-v2.vercel.app';
    console.log(`- Navigating the new tab to: ${targetUrl}`);
    await page.goto(targetUrl, { waitUntil: 'load', timeout: 30000 });
    await page.waitForTimeout(3000);

    const screenshotDir = path.join('C:', 'Users', 'mjjeon', 'Desktop', 'QMS 프로젝트', 'shinwoo-valve-qms', '안티그래비티', 'screenshot');
    if (!fs.existsSync(screenshotDir)) {
      fs.mkdirSync(screenshotDir, { recursive: true });
    }
    
    const proofPath = path.join(screenshotDir, 'cdp_new_tab_proof.png');
    await page.screenshot({ path: proofPath });
    console.log('✅ Screenshot proof captured and saved: ' + proofPath);

  } catch (err) {
    console.error('❌ CDP 새 탭 실증 실패:', err.message);
  } finally {
    if (browser) {
      await browser.close(); // 세션의 안전한 닫기 처리
    }
  }
}
verifyCdpTab();
```
